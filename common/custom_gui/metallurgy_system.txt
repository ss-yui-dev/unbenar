custom_button = {
	name = show_mettalurgy
	tooltip = show_metallurgy_tt
	potential = {
		culture_group = dwarven
	}
	trigger = {
		ai = no
	}
	effect = {
		if = {
			limit = {
				FROM = { NOT = { has_country_flag = metall_menu_open } } 
			}
			FROM = {
				set_country_flag = metall_menu_open
			}
		}
		else_if = {
			limit = {
				FROM = { has_country_flag = metall_menu_open }
			}
			FROM = {
				clr_country_flag = metall_menu_open
			}
		}
	}
}

custom_icon = {
	name = metallurgy_system
	potential = {
		has_country_flag = metall_menu_open
	}
}
