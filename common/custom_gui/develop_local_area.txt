custom_button = {
	name = develop_local_area
	potential = {
		always = yes
	}
	trigger = {
		is_state = yes
		all_province = {
			OR = {
				NOT = { area = PREV }
				AND = {
					owned_by = ROOT
					NOT = { has_province_modifier = ANEX_local_development_5 }
				}
			}
		}
	}
	effect = {
		if = {
			limit = {
				owner = {
					NOT = {
						has_idea = development_ideas7
					}
				}
			}
			if= {
				limit = {
					owner = {
						NOT = {
							monthly_income = 55
						}
					}
				}
				owner = { add_years_of_income = -0.75 }
			}
			else = {
				owner = { add_treasury = -500 }
			}
		}
		else_if = {
			limit = {
				owner = { has_idea = development_ideas7 }
			}
			if= {
				limit = {
					owner = {
						NOT = {
							monthly_income = 41
						}
					}
				}
				owner = { add_years_of_income = -0.5 }
			}
			else = {
				owner = { add_treasury = -250 }
			}
		}
		every_province_in_state = {
			limit = {
				area = PREV
				owned_by = ROOT
				NOT = { has_province_modifier = ANEX_local_development_5 }
				OR = {
					AND = {
						NOT = { has_province_modifier = ANEX_local_development_1 }
						NOT = { has_province_modifier = ANEX_local_development_2 }
						NOT = { has_province_modifier = ANEX_local_development_3 }
						NOT = { has_province_modifier = ANEX_local_development_4 }
						NOT = { has_province_modifier = ANEX_local_development_5 }
					}
					has_province_modifier = ANEX_local_development_1
					has_province_modifier = ANEX_local_development_2
					has_province_modifier = ANEX_local_development_3
					has_province_modifier = ANEX_local_development_4
				}
			}
			if = {
				limit = {
					has_province_modifier = ANEX_local_development_4
				}
				remove_province_modifier = ANEX_local_development_4
				add_province_modifier = {
					name = ANEX_local_development_5
					duration = -1
				}
			}
			else_if = {
				limit = {
					has_province_modifier = ANEX_local_development_3
				}
				remove_province_modifier = ANEX_local_development_3
				add_province_modifier = {
					name = ANEX_local_development_4
					duration = -1
				}
			}
			else_if = {
				limit = {
					has_province_modifier = ANEX_local_development_2
				}
				remove_province_modifier = ANEX_local_development_2
				add_province_modifier = {
					name = ANEX_local_development_3
					duration = -1
				}
			}
			else_if = {
				limit = {
					has_province_modifier = ANEX_local_development_1
				}
				remove_province_modifier = ANEX_local_development_1
				add_province_modifier = {
					name = ANEX_local_development_2
					duration = -1
				}
			}
			else = {
				add_province_modifier = {
					name = ANEX_local_development_1
					duration = -1
				}
			}
		}
	}
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 0
			NOT = { years_of_income = 10 }
			NOT = { treasury = 1000 }
		}
	}
	tooltip = ANEX_develop_local_area_title
}
