							change_variable = {
								which = DA_MAX_Production_Points
								value = 4
							}
							change_variable = {
								which = DA_Damestear_productions_points
								value = 4
							}
						}
					}
					else = {
						owner = {
							change_variable = {
								which = DA_MAX_Production_Points
								value = 2
							}
							change_variable = {
								which = DA_Damestear_productions_points
								value = 2
							}
						}
					}
				}
			}
			if = {
				limit = {
					any_subject_country = {
						damestear = 1
					}
				}
				every_subject_country = {
					limit = {
						is_colonial_nation = yes
					}
					every_owned_province = {
						limit = {
							trade_goods = damestear
						}
						if = {
							limit = {
								base_production = 14
							}
							root = {
								change_variable = {
									which = DA_MAX_Production_Points
									value = 6
								}
								change_variable = {
									which = DA_Damestear_productions_points
									value = 6
								}
							}
						}
						else_if = {
							limit = {
								base_production = 7
							}
							root = {
								change_variable = {
									which = DA_MAX_Production_Points
									value = 4
								}
								change_variable = {
									which = DA_Damestear_productions_points
									value = 4
								}
							}
						}
						else = {
							root = {
								change_variable = {
									which = DA_MAX_Production_Points
									value = 2
								}
								change_variable = {
									which = DA_Damestear_productions_points
									value = 2
								}
							}
						}
					}
				}
			}

			#static sources from things like missions and upkeep
			change_variable = {
				which = DA_MAX_Production_Points
				which = DA_STATIC_Production_Points
			}

			# Points Multiplier
			set_variable = {
				which = DA_Production_Points_Multiplier
				value = 1
			}

			#IF YOU WANT TO ADD MORE MULTIPLIERS MAKE SURE TO add to DA_get_production_multipliers in anb_custom_artifice_localization or the multipliers may not show up correctly

			#Multipliers from government
			if = {
				limit = {
					has_country_flag = DA_Enhanced_production_x1.25
				}
				change_variable = {
					which = DA_Production_Points_Multiplier
					value = 0.25
				}
			}
			if = {
				limit = {
					has_country_flag = DA_Enhanced_production_x1.5
				}
				subtract_variable = {
					which = DA_Production_Points_Multiplier
					value = 0.5
				}
			}
			if = {
				limit = {
					has_country_flag = DA_Enhanced_production_x2
				}
				change_variable = {
					which = DA_Production_Points_Multiplier
					value = 1
				}
			}
			
			# Floor at 0
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = DA_Production_Points_Multiplier
							value = 0
						}
					}
				}
				set_variable = {
					which = DA_Production_Points_Multiplier
					value = 0
				}
			}
			else = {
				multiply_variable = {
					which = DA_MAX_Production_Points
					which = DA_Production_Points_Multiplier
				}
			}

			# Floor at 0
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = DA_MAX_Production_Points
							value = 0
						}
					}
				}
				set_variable = {
					which = DA_MAX_Production_Points
					value = 0
				}
			}

			set_variable = {
				which = DA_REMAINING_Production_Points
				value = 0
			}

			change_variable = {
				which = DA_REMAINING_Production_Points
				which = DA_MAX_Production_Points
			}

			subtract_variable = {
				which = DA_REMAINING_Production_Points
				which = DA_Production_Points_Upkeep
			}

			# If over capacity
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = DA_REMAINING_Production_Points
							value = 0
						}
					}
				}
				add_country_modifier = { name =  DA_over_production_cap duration = -1 }
			}
			else = {
				remove_country_modifier = DA_over_production_cap
			}
		}
	}
}
