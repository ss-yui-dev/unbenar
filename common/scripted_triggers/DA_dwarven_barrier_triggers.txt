DA_can_upgrade_DRB = {
	OR = {
		AND = {
			owner = {
				has_country_flag = DA_can_build_Runic_barrier
			}
			NOT = { has_Dwarven_magic_Barrier = yes }
		}
		AND = {
			has_Dwarven_magic_Barrier = yes
			owner = {
				has_country_flag = DA_can_build_Runic_barrier_t2
			}
			NOT = { has_Dwarven_magic_Barrier_t2 = yes }
		}
		AND = {
			has_Dwarven_magic_Barrier_t2 = yes
			owner = {
				has_country_flag = DA_can_build_Runic_barrier_t3
			}
		}
	}
}
