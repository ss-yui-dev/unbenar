DA_own_iron_provinces = {
	custom_trigger_tooltip = {
		tooltip = DA_own_Iron_$amount$_tt
		check_variable = {
			which = DA_Metal_owned_Iron
			value = $amount$
		}
	}
}

DA_is_Mine_not_hold = {
	is_wasteland = no
	OR = {
		has_terrain = mountain
		has_terrain = highlands
		has_terrain = cavern
		has_province_modifier = DA_Mine_1
		has_province_modifier = DA_Mine_2
		has_province_modifier = DA_Mine_3
		has_province_modifier = DA_Mine_4
		has_province_modifier = DA_Mine_5
	}
}
DA_is_Upgradable_mine = {
	OR = {
		AND = {
			base_production = 7
			has_province_modifier = DA_Mine_1
		}
		AND = {
			base_production = 14
			has_province_modifier = DA_Mine_2
		}
		AND = {
			base_production = 21
			has_province_modifier = DA_Mine_3
		}
		AND = {
			base_production = 28
			has_province_modifier = DA_Mine_4
		}
	}
}
