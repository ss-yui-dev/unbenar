
estate_monstrous_tribes_kobold_drake_riders = {
	icon = privilege_seek_support_of_clergy
	max_absolutism = 0
	loyalty = 0.05
	influence = 0.05
	is_valid = {
		has_country_modifier = kobold_administration
		has_country_modifier = kobold_military
		has_country_modifier = mythical_cavalry_drakon
	}
	can_select = {
		has_country_modifier = kobold_administration
		has_country_modifier = kobold_military
		has_country_modifier = mythical_cavalry_drakon
	}
	can_revoke = {
		always = yes
	}
	on_granted = {
		add_country_modifier = {
			name = kobold_drake_riders_modifier
			duration = -1
		}
	}
	on_revoked = {
		remove_country_modifier = kobold_drake_riders_modifier
	}
	penalties = {
	}
	benefits = {
		amount_of_hussars = 4
	}
	
	conditional_modifier = {
		trigger = { 
			has_country_flag = unlocked_artifice_invention_medicinal_potion
			NOT = { has_country_flag = unlocked_artifice_invention_trollskin_potion }
		}
		modifier = {
			cavalry_fire = 0.5
			cavalry_shock = -0.5
		}
	}
	
	conditional_modifier = {
		trigger = { 
			has_country_flag = unlocked_artifice_invention_trollskin_potion
			NOT = { has_country_flag = unlocked_artifice_invention_fire_potion }
		}
		modifier = {
			cavalry_fire = 1
			cavalry_shock = -0.75
			assault_fort_ability = 0.25
			siege_ability = 0.10
		}
	}
	
	conditional_modifier = {
		trigger = { 
			has_country_flag = unlocked_artifice_invention_fire_potion
		}
		modifier = {
			cavalry_fire = 2
			cavalry_shock = -1
			assault_fort_ability = 0.50
			siege_ability = 0.20
		}
	}

	ai_will_do = {
		factor = 10
	}
}
