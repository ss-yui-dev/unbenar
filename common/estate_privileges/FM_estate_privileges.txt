estate_mages_new_world_mission = {
	icon = privilege_new_world_mission
	max_absolutism = -5
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 1 } }
	loyalty = 0.05
	influence = 0.05
	is_valid = {
		OR = {
			num_of_colonists = 1
			num_of_colonies = 1
		}
		NOT = { has_estate_privilege = estate_church_new_world_mission }
	}
	penalties = {
	}
	benefits = {
		native_assimilation = 0.5
	}
	on_granted = {
		if = {
			limit = { has_dlc = "Domination" }
			enable_estate_action = {
				estate_action = NEW_WORLD_MISSION_CHURCH
			}
		}
		else = {
			custom_tooltip = has_no_effects_on_granted
		}
	}
	on_invalid = {
		if = {
			limit = { has_dlc = "Domination" }
			disable_estate_action = {
				estate_action = NEW_WORLD_MISSION_CHURCH
			}
		}
		else = {
			hidden_effect = {
				disable_estate_action = {
					estate_action = NEW_WORLD_MISSION_CHURCH
				}
			}
		}
	}
	on_revoked = {
		if = {
			limit = { has_dlc = "Domination" }
			disable_estate_action = {
				estate_action = NEW_WORLD_MISSION_CHURCH
			}
		}
		else = {
			hidden_effect = {
				disable_estate_action = {
					estate_action = NEW_WORLD_MISSION_CHURCH
				}
			}
		}
	}
	conditional_modifier = {
		trigger = { has_government_mechanic = russian_modernization_mechanic }
		modifier = { monthly_russian_modernization = -0.03 }
	}
	#conditional_modifier = {
	#	trigger = { has_government_mechanic = ottoman_decadence_mechanic }
	#	modifier = { monthly_ottoman_decadence = 0.02 }
	#}
	ai_will_do = {
		factor = 0
	}
}
